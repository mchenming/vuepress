# 解决vuex页面刷新数据丢失问题（数据持久性） 默认刷新数据丢失

vuex 的使用方法 this.$store.commit('function' , payload) this.$store.getters('function',payload) this.$store.dispatch('function',payload) == > action: this.$store.commit('function',payload) 执行异步操作

默认将state里面的数据存储到localStore里面 页面刷新在获取
npm i vuex-along -S
import vuexAlong from 'vuex-along'
然后在vuex实例里面添加一个plugins:[vuexAling()]

import Vue from 'vue'
import Vuex from 'vuex'
import vuexalong from 'vuex-along'

Vue.use(Vuex)  // 默认会执行install方法

export default new Vuex.Store({
  state: {
    age: 15,
    name: 'chenm'
  },
  plugins:[vuexalong()],
  mutations: {
    // 派发调用时执行
    syncChange(state,changedaya){
      state.age +=changedaya
    },
    jianfaChange(state,payload){
      state.age = state.age - payload
    }
  },
  getters: {
    // 获取时候执行执行一次
    myAge(state){
      return state.age - 1
    },
    ishold (state) {
      return state.name + 'ai'
    }
  },
  actions: {
    asyncAdd(store,payload){
      // 异步方法执行异步 $store.dispatch派发任务
      setTimeout(()=>{
        store.commit('syncChange',payload)
      },1000)
    },
    asyncjian(store,payload){
      setTimeout(()=>{
        store.commit('jianfaChange',payload)
      },100)
    }
  }
})
