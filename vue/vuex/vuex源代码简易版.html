class Store {  // 用户获取的是这个Store类的实例
    // 当用户new实例时传入的所有属性
    constructor(options){
        // console.log(options)
        // options.state
        this.vm = new Vue({  // 创建vue实例保证状态更新可以刷新试图
            data() {     // 默认这个状态会被Object.defineProperty(es5)重新定义
                return {
                    Vmstate: options.state
                }
            },
        })
        // console.log(this) => Store
        let getters = options.getters
        this.getters = {}
        Object.keys(getters).forEach(key=>{
            Object.defineProperty(this.getters,key,{   //当数据变执行
                get:()=>{
                    // todo...
                    return getters[key](this.state)
                },
                // set:(val)=>{
                //     val = getters[key]
                // }
            })
        })

        // 将用户定义的mutation放到store上
        let mutations = options.mutations
        this.mutations = {}
        Object.keys(mutations).forEach(key => {
            // Object.defineProperty(this.mutations,key,{
            //     get:()=>{
            //         return mutations[key](this.state,payload)
            //     }
            // })
            this.mutations[key] = (payload)=> {  // 订阅   //一个方法function
                mutations[key](this.state,payload)
            }
        })

        let actions = options.actions
        this.actions = {}
        Object.keys(actions).forEach(key => {
            this.actions[key] = (payload)=>{  // 一个方法function
                actions[key](this,payload)
            } 
        })
    }
    // this.$store.commit(functionName,payload)
    commit = (mutationsName,payload) => { //es7的写法  这个里面的this永远指向当前store的实例
        this.mutations[mutationsName](payload)  //发布
    }
    dispatch = (actionsName,payload) =>{
        this.actions[actionsName](payload)
    }
    get state(){ //获取实例上的Vmstate属性就会执行此方法   只要Vmstate变化就执行（es6的类访问器）
      return this.vm.Vmstate       
    }
}

// install注册插件
const install = (_Vue) =>{  // vue的构造函数
    Vue = _Vue      // vue的构造函数
    // 放在vue的原型上不对 因为默认会给所有的实例增加
    // 只从当前的根实例开始所有跟实例的子组件才有$store方法
    Vue.mixin({     // 组件的创建过程是先父后子
        beforeCreate() {
            //把父组件的store属性放到每一个组件的实例上
            // console.log(this.$root)
            if(this.$options.store){
                //根组件main.js
                // console.log(this.$options.name)
                this.$store = this.$options.store
            } else {
                // app.vue
                this.$store = this.$parent && this.$parent.$store
            }
        },
    })  // 混入的生命周期先于组件的生命周期
}

export default {
    install,
    Store
}
