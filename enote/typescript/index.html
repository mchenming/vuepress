# class类属性(调用new plMethod 默认执行constructor的方法)

class plMethod {
    saveName: string;
    id: number;
    constructor (saveName:string) {
        this.saveName = saveName
        this.id = 0
    }
    // 获取本地存储的信息
    readList():any  {
        let str: string | null = localStorage.getItem(this.saveName)
        let arr: Array<object>
        if(str !== null){
            arr = JSON.parse(str);
        } else {
            arr = []
        }
        return arr
    }
    // 更新存数据
    savaList(savearr:Array<object>):void {
        let str:string = JSON.stringify(savearr)
        localStorage.setItem(this.saveName,str)
    }
    // 追加数据进去
    addList(contentstr:string):void {
        // console.log(123)
        // console.log(contentstr)
        let arr = this.readList()
        let obj:{content:string,id: number} = {
          content : contentstr,
          id : this.id + 1
        };
        arr.push(obj)
        this.savaList(arr)
    }
    // 删除评论
    removeList(id: number){
        let arr = this.readList()
        let index: number = arr.findIndex((ele:any) => {
            return ele.id === id
        })
        if(index>-1){
            arr.splice(index,1)
            this.savaList(arr)
        }
    }
}

    let fun = new plMethod('pltest')
    let arr = fun.readList()

    function as(arr){
        let str = ''
        arr.forEach(item=>{
            str += '<div style="display:flex;justify-content:space-between;padding:0 20px;margin:5px 0;"><div>'+item.content+'</div><button attr='+item.id+' onclick="abc('+item.id+')">删除</button></div>'
