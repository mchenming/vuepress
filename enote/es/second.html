// 数组类型
let oldProto = Array.prototype
let proto = Object.create(oldProto)
var wayarr = ['shift','push']
wayarr.forEach(item=>{
    proto[item] = function (){
        update()
        oldProto[item].apply(this,arguments)
    }
})
function observer(value){  // 观察
    if(typeof value !== 'object'){
        return value
    }
    if(Array.isArray(value)){
        return value._proto_ = proto
    }
    for(let key in value){
        defineReactive(value,key,value[key])
    }
}

function defineReactive(obj,key,item){
    observer(item)
    Object.defineProperty(obj,key,{
        get(){
            return item
        },
        set(val){
            if(item!==val){
                observer(val)
                item = val
                update()
            }
        }
    })
}

observer(obj)

obj.name = {n:200}
